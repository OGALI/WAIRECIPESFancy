- <ListItem>1 can chickpeas, drained</ListItem>
- <ListItem>1 garlic clove</ListItem>
- <ListItem>1/3 cup tahini</ListItem>
- <ListItem>1/3 cup olive oil</ListItem>
- <ListItem>1 lemon, zested and 1Tbsp lemon juice</ListItem>
- <ListItem>1 tsp salt</ListItem>
